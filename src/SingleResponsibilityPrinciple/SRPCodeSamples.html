public static double Median(double[] numbers)
{
	var sortedArray = sortArray(numbers);

	double median;

	var middleIndex = numbers.Length / 2;

	if (numbers.Length % 2 == 0)
	{
		var middleItemsSum = sortedArray[middleIndex - 1] + sortedArray[middleIndex];

		median = middleItemsSum / 2.0;
	}
	else
		median = sortedArray[middleIndex];

	return median;
}

public static double[] Diff(double[] arrayA, double[] arrayB)
{
	double[] diffArray;
	int length;
	if (arrayA== null || ((length= arrayA.Length)<1)||
		arrayB== null || ((length= arrayB.Length)<1)||
		arrayA.Length!= arrayB.Length)
		return null;

	diffArray = new double[length];

	for (int i = 0; i<length; ++i)
		diffArray[i]= arrayA[i]- arrayB[i];

	return diffArray;

public static double[] Diff(
	double[] a,
	double[] b)
{
	if (!AreValid(a, b))
		return null;

	return SubtractArrays(a, b);
}

private static bool AreValid(
	double[] a,
	double[] b)
{
	return HasItems(a)
		   && HasItems(b)
		   && HasSameLength(a, b);
}

private static bool HasSameLength(
	double[] a,
	double[] b)
{
	return a.Length == b.Length;
}

public static bool HasItems(
	double[] a)
{
	if (a == null)
		return false;

	return a.Length > 0;
}

private static double[] SubtractArrays(
	double[] a,
	double[] b)
{
	var diff = new double[a.Length];

	for (var i = 0; i < a.Length; ++i)
		diff[i] = a[i] - b[i];

	return diff;
}
